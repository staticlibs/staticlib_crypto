147  const std::string& get_hash() {
148  if (hash.empty()) {

151  if (1 != err) throw crypto_exception(TRACEMSG(

153  auto dest = sl::io::string_sink();
154  {
155  auto bsrc = sl::io::array_source(reinterpret_cast<const char*>(buf.data()), buf.size());
156  auto hex = sl::io::make_hex_sink(dest);
157  sl::io::copy_all(bsrc, hex);
158  }
159  hash = std::move(dest.get_string());
