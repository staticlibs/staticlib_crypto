73  explicit base64_sink(Sink&& sink) :
74  sink(std::move(sink)),
75  b64(BIO_new(BIO_f_base64()), [](BIO* bio) { BIO_free(bio); }),
76  bsrc(BIO_new(BIO_s_bio()), [](BIO* bio) { BIO_free(bio); }),
77  bsink(BIO_new(BIO_s_bio()), [](BIO* bio) { BIO_free(bio); }) {
78  if (nullptr == b64.get()) throw crypto_exception(TRACEMSG(
79  "'BIO_new(BIO_f_base64)' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
80  if (nullptr == bsrc.get() || nullptr == bsink.get()) throw crypto_exception(TRACEMSG(
81  "'BIO_new(BIO_s_bio)' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
82  // base64 format
83  BIO_set_flags(b64.get(), BIO_FLAGS_BASE64_NO_NL);
84  // source
85  int err_src_buf_size = BIO_set_write_buf_size(bsrc.get(), buffer_size);
86  if(1 != err_src_buf_size) throw crypto_exception(TRACEMSG(
87  "'BIO_set_write_buf_size' error, size: [" + sl::support::to_string(buffer_size) + "]," +
88  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
89  // sink
90  int err_sink_buf_size = BIO_set_write_buf_size(bsink.get(), buffer_size);
91  if(1 != err_sink_buf_size) throw crypto_exception(TRACEMSG(
92  "'BIO_set_write_buf_size' error, size: [" + sl::support::to_string(buffer_size) + "]," +
93  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
94  // chain
95  BIO* pushed = BIO_push(b64.get(), bsrc.get());
96  if(pushed != b64.get()) throw crypto_exception(TRACEMSG(
97  "'BIO_push' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
98  // pair, BIO_s_mem may be used instead
99  int err_pair = BIO_make_bio_pair(bsrc.get(), bsink.get());
100  if (1 != err_pair) throw crypto_exception(TRACEMSG(
101  "'BIO_make_bio_pair' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
102  }

163  std::streamsize write(sl::io::span<const char> span) {
164  size_t written = 0;
165  while (written < span.size()) {
166  // write
167  auto allowed = BIO_get_write_guarantee(b64.get());
168  if (allowed <= 0) throw crypto_exception(TRACEMSG(
169  "'BIO_get_write_guarantee' write buffer overflow," +
170  " allowed: [" + sl::support::to_string(allowed) + "]," +
171  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
172  auto uallowed = static_cast<size_t>(allowed);
173  auto avail = span.size() - written;
174  size_t to_write = avail <= uallowed ? avail : uallowed;
175  auto wr = BIO_write(b64.get(), span.data() + written, static_cast<int>(to_write));
176  if (wr <= 0) throw crypto_exception(TRACEMSG(
177  "'BIO_write' error, to_write: [" + sl::support::to_string(to_write) + "]," +
178  " written: [" + sl::support::to_string(wr) + "]," +
179  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
180  written += static_cast<size_t>(wr);
181  // read and write to sink
182  int read = 0;
183  while ((read = (BIO_read(bsink.get(), buf.data(), static_cast<int>(buf.size())))) > 0) {
184  sl::io::write_all(sink, {buf.data(), static_cast<size_t>(read)});
185  }
186  if (read < -1) throw crypto_exception(TRACEMSG(
187  "'BIO_read' error, return: [" + sl::support::to_string(read) + "]," +
188  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
189  }
190  return span.size_signed();
191  }

198  std::streamsize flush() {
199  auto err = BIO_flush(b64.get());
200  if (1 != err) throw crypto_exception(TRACEMSG(
201  "'BIO_flush' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
202  // read and write to sink
203  std::streamsize written = 0;
204  int read = 0;
205  while ((read = (BIO_read(bsink.get(), buf.data(), static_cast<int>(buf.size())))) > 0) {
206  written += read;
207  sl::io::write_all(sink, {buf.data(), static_cast<size_t>(read)});
208  }
209  if (read < -1) throw crypto_exception(TRACEMSG(
210  "'BIO_read' error, return: [" + sl::support::to_string(ERR_get_error()) + "]"));
211  sink.flush();
212  return written;
213  }
