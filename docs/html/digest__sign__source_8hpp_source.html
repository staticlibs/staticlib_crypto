175  const std::string& get_signature() {
176  if (signature.empty()) {
177  size_t req = 0;
178  auto err_req = EVP_DigestSignFinal(ctx.get(), nullptr, std::addressof(req));
179  if (1 != err_req || req <= 0) throw crypto_exception(TRACEMSG(
180  "'EVP_DigestSignFinal' req error, code: [" + sl::support::to_string(ERR_get_error()) + "]," +
181  " req: [" + sl::support::to_string(req) + "]"));
182  std::string sig;
183  sig.resize(req);
184  size_t slen = req;
185  auto err = EVP_DigestSignFinal(ctx.get(), reinterpret_cast<unsigned char*>(std::addressof(sig.front())),
186  std::addressof(slen));
187  if (1 != err || req != slen) throw crypto_exception(TRACEMSG(
188  "'EVP_DigestSignFinal' error, code: [" + sl::support::to_string(ERR_get_error()) + "]," +
189  " req: [" + sl::support::to_string(req) + "]," +
190  " slen: [" + sl::support::to_string(slen) + "]"));
191  signature = sl::io::string_to_hex(sig);
192  }
