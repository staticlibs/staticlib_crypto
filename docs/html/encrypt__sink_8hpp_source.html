73  encrypt_sink(Sink&& sink, const EVP_CIPHER* cipher, const std::string& key, const std::string& iv) :
74  sink(std::move(sink)),
75  encrypter(BIO_new(BIO_f_cipher()), [](BIO* bio) { BIO_free(bio); }),
76  bsrc(BIO_new(BIO_s_bio()), [](BIO* bio) { BIO_free(bio); }),
77  bsink(BIO_new(BIO_s_bio()), [](BIO* bio) { BIO_free(bio); }) {
78  if (nullptr == encrypter.get()) throw crypto_exception(TRACEMSG(
79  "'BIO_new(BIO_f_cipher)' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
80  if (nullptr == bsrc.get() || nullptr == bsink.get()) throw crypto_exception(TRACEMSG(
81  "'BIO_new(BIO_s_bio)' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
82  BIO_set_cipher(encrypter.get(), cipher, reinterpret_cast<const unsigned char*>(key.c_str()),
83  reinterpret_cast<const unsigned char*>(iv.c_str()), 1 /* encrypt */);
84  EVP_CIPHER_CTX* pctx = nullptr;
85  auto err_ctx = BIO_get_cipher_ctx(encrypter.get(), std::addressof(pctx));
86  if (1 != err_ctx || nullptr == pctx) throw crypto_exception(TRACEMSG(
87  "'BIO_set_cipher' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
88  // source
89  int err_src_buf_size = BIO_set_write_buf_size(bsrc.get(), buffer_size);
90  if(1 != err_src_buf_size) throw crypto_exception(TRACEMSG(
91  "'BIO_set_write_buf_size' error, size: [" + sl::support::to_string(buffer_size) + "]," +
92  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
93  // sink
94  int err_sink_buf_size = BIO_set_write_buf_size(bsink.get(), buffer_size);
95  if(1 != err_sink_buf_size) throw crypto_exception(TRACEMSG(
96  "'BIO_set_write_buf_size' error, size: [" + sl::support::to_string(buffer_size) + "]," +
97  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
98  // chain
99  BIO* pushed = BIO_push(encrypter.get(), bsrc.get());
100  if(pushed != encrypter.get()) throw crypto_exception(TRACEMSG(
101  "'BIO_push' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
102  // pair, BIO_s_mem may be used instead
103  int err_pair = BIO_make_bio_pair(bsrc.get(), bsink.get());
104  if (1 != err_pair) throw crypto_exception(TRACEMSG(
105  "'BIO_make_bio_pair' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
106  }

167  std::streamsize write(sl::io::span<const char> span) {
168  size_t written = 0;
169  while (written < span.size()) {
170  // write
171  auto allowed = BIO_get_write_guarantee(encrypter.get());
172  if (allowed <= 0) throw crypto_exception(TRACEMSG(
173  "'BIO_get_write_guarantee' write buffer overflow," +
174  " allowed: [" + sl::support::to_string(allowed) + "]," +
175  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
176  auto uallowed = static_cast<size_t>(allowed);
177  auto avail = span.size() - written;
178  size_t to_write = avail <= uallowed ? avail : uallowed;
179  auto wr = BIO_write(encrypter.get(), span.data() + written, static_cast<int>(to_write));
180  if (wr <= 0) throw crypto_exception(TRACEMSG(
181  "'BIO_write' error, to_write: [" + sl::support::to_string(to_write) + "]," +
182  " written: [" + sl::support::to_string(wr) + "]," +
183  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
184  written += static_cast<size_t>(wr);
185  // read and write to sink
186  int read = 0;
187  while ((read = (BIO_read(bsink.get(), buf.data(), static_cast<int>(buf.size())))) > 0) {
188  sl::io::write_all(sink, {buf.data(), static_cast<size_t>(read)});
189  }
190  if (read < -1) throw crypto_exception(TRACEMSG(
191  "'BIO_read' error, return: [" + sl::support::to_string(read) + "]," +
192  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
193  }
194  return span.size_signed();
195  }

202  std::streamsize flush() {
203  auto err = BIO_flush(encrypter.get());
204  if (1 != err) throw crypto_exception(TRACEMSG(
205  "'BIO_flush' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
206  // read and write to sink
207  std::streamsize written = 0;
208  int read = 0;
209  while ((read = (BIO_read(bsink.get(), buf.data(), static_cast<int>(buf.size())))) > 0) {
210  written += read;
211  sl::io::write_all(sink, {buf.data(), static_cast<size_t>(read)});
212  }
213  if (read < -1) throw crypto_exception(TRACEMSG(
214  "'BIO_read' error, return: [" + sl::support::to_string(read) + "]," +
215  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
216  sink.flush();
217  return written;
218  }
