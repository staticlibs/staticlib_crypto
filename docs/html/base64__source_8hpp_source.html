153  std::streamsize read(sl::io::span<char> span) {
154  // read buffered
155  int read_buffered = BIO_read(b64.get(), span.data(), static_cast<int>(span.size_signed()));
156  if (read_buffered > 0) {
157  return static_cast<std::streamsize>(read_buffered);
158  }
159  if (read_buffered < -1) throw crypto_exception(TRACEMSG(
160  "'BIO_read' buffered error, return: [" + sl::support::to_string(read_buffered) + "]," +
161  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
162  // decode more data
163  auto allowed = BIO_get_write_guarantee(bsrc.get());
164  if (allowed <= 0) throw crypto_exception(TRACEMSG(
165  "'BIO_get_write_guarantee' write buffer overflow," +
166  " allowed: [" + sl::support::to_string(allowed) + "]," +
167  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
168  auto uallowed = static_cast<size_t>(allowed);
169  size_t to_decode = uallowed <= buf.size() ? uallowed : buf.size();
170  size_t read_from_src = sl::io::read_all(src, {buf.data(), to_decode});
171  if (0 == read_from_src) {
172  // source exhausted
173  auto err_flush = BIO_flush(bsrc.get());
174  if (1 != err_flush) throw crypto_exception(TRACEMSG(
175  "'BIO_flush' error, code: [" + sl::support::to_string(err_flush) + "]," +
176  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
177  // read and return
178  int read_flushed = BIO_read(b64.get(), span.data(), static_cast<int>(span.size_signed()));
179  if (read_flushed < -1) throw crypto_exception(TRACEMSG(
180  "'BIO_read' error, return: [" + sl::support::to_string(read_flushed) + "]," +
181  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
182  return read_flushed >= 0 ? static_cast<std::streamsize>(read_flushed) : std::char_traits<char>::eof();
183  }
184  auto written = BIO_write(bsrc.get(), buf.data(), static_cast<int>(read_from_src));
185  if (written <= 0) throw crypto_exception(TRACEMSG(
186  "'BIO_write' error, to_write: [" + sl::support::to_string(read_from_src) + "]," +
187  " written: [" + sl::support::to_string(written) + "]," +
188  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
189  if (read_from_src < to_decode) {
190  // source exhausted
191  auto err_flush = BIO_flush(bsrc.get());
192  if (1 != err_flush) throw crypto_exception(TRACEMSG(
193  "'BIO_flush' error, code: [" + sl::support::to_string(err_flush) + "]," +
194  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
195  }
196  // read and return
197  int read_decoded = BIO_read(b64.get(), span.data(), static_cast<int>(span.size_signed()));
198  if (read_decoded < -1) throw crypto_exception(TRACEMSG(
199  "'BIO_read' error, return: [" + sl::support::to_string(read_decoded) + "]," +
200  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
201  return read_decoded >= 0 ? static_cast<std::streamsize>(read_decoded) : 0;
202  }
