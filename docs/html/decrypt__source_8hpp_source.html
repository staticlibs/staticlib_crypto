78  "'BIO_new(BIO_f_cipher)' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));

157  std::streamsize read(sl::io::span<char> span) {
158  // read buffered
159  int read_buffered = BIO_read(decrypter.get(), span.data(), static_cast<int>(span.size_signed()));
160  if (read_buffered > 0) {
161  return static_cast<std::streamsize>(read_buffered);
162  }
163  if (read_buffered < -1) throw crypto_exception(TRACEMSG(
164  "'BIO_read' buffered error, return: [" + sl::support::to_string(read_buffered) + "]," +
165  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
166  // decode more data
167  auto allowed = BIO_get_write_guarantee(bsrc.get());
168  if (allowed <= 0) throw crypto_exception(TRACEMSG(
169  "'BIO_get_write_guarantee' write buffer overflow," +
170  " allowed: [" + sl::support::to_string(allowed) + "]," +
171  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
172  auto uallowed = static_cast<size_t>(allowed);
173  size_t to_decode = uallowed <= buf.size() ? uallowed : buf.size();
174  size_t read_from_src = sl::io::read_all(src, {buf.data(), to_decode});
175  if (0 == read_from_src) {
176  // source exhausted
177  auto err_flush = BIO_flush(bsrc.get());
178  if (1 != err_flush) throw crypto_exception(TRACEMSG(
179  "'BIO_flush' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
180  unsigned char eof = static_cast<unsigned char>(-1);
181  auto written_flushed = BIO_write(bsrc.get(), std::addressof(eof), 1);
182  if (written_flushed <= 0) throw crypto_exception(TRACEMSG(
183  "'BIO_write' eof error, to_write: [" + sl::support::to_string(read_from_src) + "]," +
184  " written: [" + sl::support::to_string(written_flushed) + "]," +
185  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
186  // read and return
187  int read_flushed = BIO_read(decrypter.get(), span.data(), static_cast<int>(span.size_signed()));
188  if (read_flushed < -1) throw crypto_exception(TRACEMSG(
189  "'BIO_read' error, read: [" + sl::support::to_string(read_flushed) + "]," +
190  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
191  int read_status = BIO_get_cipher_status(decrypter.get());
192  if (1 != read_status) throw crypto_exception(TRACEMSG(
193  "'BIO_get_cipher_status' error, read: [" + sl::support::to_string(read_flushed) + "]," +
194  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
195  return read_flushed >= 0 ? static_cast<std::streamsize>(read_flushed) : std::char_traits<char>::eof();
196  }
197  auto written = BIO_write(bsrc.get(), buf.data(), static_cast<int>(read_from_src));
198  if (written <= 0) throw crypto_exception(TRACEMSG(
199  "'BIO_write' error, to_write: [" + sl::support::to_string(read_from_src) + "]," +
200  " written: [" + sl::support::to_string(written) + "]," +
201  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
202  if (read_from_src < to_decode) {
203  // source exhausted
204  auto err_flush = BIO_flush(bsrc.get());
205  if (1 != err_flush) throw crypto_exception(TRACEMSG(
206  "'BIO_flush' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
207  unsigned char eof = static_cast<unsigned char>(-1);
208  auto written_flushed = BIO_write(bsrc.get(), std::addressof(eof), 1);
209  if (written_flushed <= 0) throw crypto_exception(TRACEMSG(
210  "'BIO_write' eof error, to_write: [" + sl::support::to_string(read_from_src) + "]," +
211  " written: [" + sl::support::to_string(written_flushed) + "]," +
212  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
213  }
214  // read and return
215  int read_decoded = BIO_read(decrypter.get(), span.data(), static_cast<int>(span.size_signed()));
216  if (read_decoded < -1) throw crypto_exception(TRACEMSG(
217  "'BIO_read' error, return: [" + sl::support::to_string(read_decoded) + "]," +
218  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
219  int read_status = BIO_get_cipher_status(decrypter.get());
220  if (1 != read_status) throw crypto_exception(TRACEMSG(
221  "'BIO_get_cipher_status' error, read: [" + sl::support::to_string(read_decoded) + "]," +
222  " code: [" + sl::support::to_string(ERR_get_error()) + "]"));
223  return read_decoded >= 0 ? static_cast<std::streamsize>(read_decoded) : 0;
224  }
