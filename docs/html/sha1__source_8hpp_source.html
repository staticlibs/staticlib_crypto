137  const std::string& get_hash() {
138  if (hash.empty()) {
139  std::array<unsigned char, SHA_DIGEST_LENGTH> buf;
140  auto err = SHA1_Final(buf.data(), ctx.get());
141  if (1 != err) throw crypto_exception(TRACEMSG(
142  "'SHA1_Final' error, code: [" + sl::support::to_string(ERR_get_error()) + "]"));
143  auto dest = sl::io::string_sink();
144  {
145  auto bsrc = sl::io::array_source(reinterpret_cast<const char*>(buf.data()), buf.size());
146  auto hex = sl::io::make_hex_sink(dest);
147  sl::io::copy_all(bsrc, hex);
148  }
149  hash = std::move(dest.get_string());
150  }
151  return hash;
